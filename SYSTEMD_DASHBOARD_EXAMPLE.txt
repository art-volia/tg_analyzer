# /etc/systemd/system/tg-dashboard.service
[Unit]
Description=Telegram Analyzer Dashboard
After=network.target

[Service]
User=ubuntu
WorkingDirectory=/home/ubuntu/tg-analyzer
ExecStart=/home/ubuntu/tg-analyzer/.venv/bin/streamlit run dashboard_app.py --server.address 0.0.0.0 --server.port 8501
Environment=STREAMLIT_BROWSER_GATHER_USAGE_STATS=false
Restart=always
RestartSec=5

[Install]
WantedBy=multi-user.target
